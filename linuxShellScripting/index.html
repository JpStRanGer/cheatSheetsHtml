Linux Shell Scripting Cheat Sheet

Basics

Shebang

# The first line of a script that specifies the interpreter
#!/bin/bash

Comments

# Single line comment

: '
Multiline comment
spanning multiple lines
'

Running a Script

# Make the script executable
chmod +x script.sh

# Run the script
./script.sh

# Execute the script with bash
bash script.sh

Variables

Defining Variables

# Assign a value to a variable
variable_name="value"

# Read-only variable
readonly variable_name="value"

Accessing Variables

# Access the value of a variable
echo $variable_name
echo ${variable_name}  # Recommended for clarity

Special Variables

# Script name
$0

# Positional parameters
$1, $2, ...

# Number of arguments
$#

# All arguments as separate words
$@

# All arguments as a single word
$*

# Exit status of the last command
$?

# Process ID of the script
$$

# Process ID of the last background command
$!

Control Structures

If-Else

if [ condition ]; then
    # Commands
elif [ condition ]; then
    # Commands
else
    # Commands
fi

Case

case "$variable" in
    "pattern1")
        # Commands
        ;;
    "pattern2")
        # Commands
        ;;
    *)
        # Default commands
        ;;
esac

Loops

for i in {1..5}; do
    echo "Iteration $i"
done

count=1
while [ $count -le 5 ]; do
    echo "Iteration $count"
    count=$((count + 1))
done
